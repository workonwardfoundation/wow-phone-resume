// src/services/conversationService.js
import { getDbClient } from '../config/db.js';
import { appConfig } from '../config/app.js';
import { logger } from '../utils/logger.js';

/**
 * Saves conversation data to MongoDB.
 * @param {object} session - Session data containing conversationLog and metadata.
 * @returns {Promise<boolean>}
 */
export async function saveConversation(session) {
  if (!session || !session.conversationLog || session.conversationLog.length === 0) {
    logger.info('No conversation data to save');
    return false;
  }
  try {
    const client = getDbClient();
    // Ensure connection is alive
    await client.db(appConfig.dbName).command({ ping: 1 });
    const db = client.db(appConfig.dbName);
    const collection = db.collection(appConfig.collectionName);
    const result = await collection.insertOne({
      id: Date.now(),
      phone_number: session.phoneNumber || 'Unknown',
      stream_id: session.streamSid,
      transcript: session.conversationLog,
      timestamp: new Date().toISOString(),
    });
    logger.info(`Successfully saved conversation to MongoDB with ID: ${result.insertedId}`);
    return true;
  } catch (error) {
    logger.error('Error saving conversation to MongoDB:', error);
    return false;
  }
}
