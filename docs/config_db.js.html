// src/config/db.js
import { MongoClient } from 'mongodb';
import { appConfig } from './app.js';
import { logger } from '../utils/logger.js';

const client = new MongoClient(appConfig.mongodbUri, {
  useNewUrlParser: true,
  useUnifiedTopology: true,
});

/**
 * Connects to MongoDB with retry mechanism.
 * @param {number} retryAttempt - Current retry attempt.
 * @param {number} maxRetries - Maximum number of retries.
 */
export async function connectToDatabase(retryAttempt = 0, maxRetries = 5) {
  try {
    await client.connect();
    logger.info('Connected to MongoDB!');
    await client.db(appConfig.dbName).command({ ping: 1 });
    logger.info('MongoDB connection verified successfully!');
  } catch (error) {
    logger.error(`MongoDB connection error (attempt ${retryAttempt + 1}/${maxRetries}):`, error);
    if (retryAttempt < maxRetries) {
      const delay = Math.min(1000 * Math.pow(2, retryAttempt), 30000);
      logger.info(`Retrying in ${delay / 1000} seconds...`);
      await new Promise(resolve => setTimeout(resolve, delay));
      return connectToDatabase(retryAttempt + 1, maxRetries);
    } else {
      logger.error('Failed to connect to MongoDB after maximum retry attempts.');
      process.exit(1);
    }
  }
}

export function getDbClient() {
  return client;
}
