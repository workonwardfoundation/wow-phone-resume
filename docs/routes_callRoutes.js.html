// src/routes/callRoutes.js
import { handleIncomingCall, handleLanguageSelection } from '../controllers/callController.js';

/**
 * Registers call-related routes.
 * @param {object} fastify - The Fastify instance.
 * @param {object} options - Plugin options.
 */
export default async function callRoutes(fastify, options) {
  fastify.all('/incoming-call', async (request, reply) => {
    const phoneNumber =
      request.body?.From ||
      request.query?.From ||
      request.headers['x-twilio-from'] ||
      "Unknown";
    const response = handleIncomingCall(phoneNumber);
    reply.type('text/xml').send(response);
  });

  fastify.all('/language-selection', async (request, reply) => {
    const digits = request.body?.Digits || request.query?.Digits;
    const phoneNumber = request.body?.From || request.query?.From || "Unknown";
    const host = request.headers.host; // dynamically retrieve the host (ngrok URL)
    const response = handleLanguageSelection(digits, phoneNumber, host);
    reply.type('text/xml').send(response);
  });
}
